                .url(BASE_URL + "/chat/completions")
                .addHeader("Authorization", "Bearer " + API_KEY)
                .addHeader("Content-Type", "application/json")

                        MediaType.parse("application/json; charset=utf-8"),
                        requestBody.toString()))
                .build();

        try (Response response = client.newCall(request).execute()) {
            if (response.isSuccessful() && response.body() != null) {

                System.out.println(content);

                System.out.println("请求失败: " + response.code() + " - " + response.message());

## 集成到Spring Boot，彻底脱离Key管理

如果你的项目用了Spring Boot，集成过程会更优雅。这里用配置类来演示极度简洁的集成：

java
@Configuration
public class AiConfig {
    
    @Value("${yunwuai.api.key}")
    private String apiKey;
    
    @Bean
    public OkHttpClient httpClient() {
        return new OkHttpClient.Builder()
                .addInterceptor(chain -> {
                    Request original = chain.request();
                    Request authorized = original.newBuilder()
                            .header("Authorization", "Bearer " + apiKey)
                            .header("Content-Type", "application/json")
                            .build();
                    return chain.proceed(authorized);
                })
                .connectTimeout(60, TimeUnit.SECONDS)
                .readTimeout(120, TimeUnit.SECONDS)
                .build();
    }
    
    @Bean
    public AiService aiService(OkHttpClient client) {
        return new AiService(client, "https://www.yunwuai.cc/v1");
    }
}
