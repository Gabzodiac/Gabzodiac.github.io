# -*- coding: utf-8 -*-
from openai import OpenAI
from typing import Optional

# 配置你的API密钥和云雾API地址
API_KEY = "sk-你的密钥" # 替换成你在[[云雾ai官网](https://www.yunwuai.cc/)](https://www.yunwuai.cc/)申请的密钥
BASE_URL = "https://www.yunwuai.cc/v1" # 这是云雾api官网提供的OpenAI兼容接口地址

def call_glm45_stream(prompt: str, temperature: float = 0.7) -> Optional[str]:
    """
    流式调用GLM-4.5模型，支持流式输出。
    Args:
        prompt: 用户输入的提示词
        temperature: 温度参数，控制输出的随机性，范围0-2。
    Returns:
        模型返回的完整回复字符串，如果报错则返回None。
    """
    full_response = ""
    client = OpenAI(api_key=API_KEY, base_url=BASE_URL)
    
    try:
        # 创建流式请求
        response = client.chat.completions.create(
            model="glm-4-plus", # 在[[云雾ai官网](https://www.yunwuai.cc/)](https://www.yunwuai.cc/)中，GLM-4.5的模型ID是glm-4-plus
            messages=[{"role": "user", "content": prompt}],
            temperature=temperature,
            stream=True # 开启流式输出，带来极致的响应体验
        )

        print("\n🤖 GLM-4.5 回复：", end="", flush=True)
        for chunk in response:
            # 从流式响应块中提取增量文本
            if chunk.choices[0].delta.content is not None:
                content_chunk = chunk.choices[0].delta.content
                print(content_chunk, end="", flush=True) # 实时打印
                full_response += content_chunk

        print("\n" + "-" * 30)
        return full_response

    except Exception as e:
        print(f"\n❌ 调用失败：{e}")
        return None

# 当直接运行此脚本时
if __name__ == "__main__":
    # 测试多轮对话或复杂推理
    test_prompt = """
    请用Python实现一个简单的斐波那契数列生成器，要求：
    1. 包含递归和迭代两种实现方式
    2. 在每个函数上方添加中文注释
    3. 打印前15个斐波那契数
    """
    result = call_glm45_stream(prompt=test_prompt)
    if result:
        print("✅ 调用成功！")


**代码核心细节拆解：**

- `BASE_URL`：这行是连接[[云雾ai官网](https://www.yunwuai.cc/)](https://www.yunwuai.cc/)的关键。你只需要把这一行从 `https://api.openai.com/v1` 改成 `https://www.yunwuai.cc/v1`。
- `model`：在[[云雾ai官网](https://www.yunwuai.cc/)](https://www.yunwuai.cc/)的模型列表中，GLM-4.5对应的是`glm-4-plus`。这个配置很灵活，如果你想用其他模型，比如DeepSeek-R1，只改这个参数就行。
- `stream=True`：流式输出能让模型更快速地返回第一个字，提升用户体验，强烈推荐开启。
