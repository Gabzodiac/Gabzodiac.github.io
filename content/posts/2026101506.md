import com.theokanning.openai.completion.chat.*;

import java.time.Duration;
import java.util.Arrays;

public class DeepSeekCoderDemo {
    public static void main(String[] args) {
        // 1. 创建OpenAiService实例，base_url指向云雾API
        OpenAiService service = new OpenAiService(
            "sk-your-yunwu-API-key-here", // 你在云雾申请的Key
            Duration.ofSeconds(60)
        );

        // 2. 设置API基础地址（核心改动点）
        service.setBaseUrl("https://www.yunwuai.cc/v1");

        // 3. 构建请求消息
        ChatCompletionRequest request = ChatCompletionRequest.builder()
            .model("deepseek-coder") // 指定DeepSeekCoder模型
            .messages(Arrays.asList(
                new ChatMessage(ChatMessageRole.SYSTEM.value(), "你是一个Java代码生成助手"),
                new ChatMessage(ChatMessageRole.USER.value(), "请写一个二分查找算法，用Java实现")
            ))
            .maxTokens(2000)
            .temperature(0.7)
            .build();

        // 4. 发送请求并获取回复
        service.createChatCompletion(request)
            .getChoices()
            .forEach(choice -> System.out.println(choice.getMessage().getContent()));
    }
}
