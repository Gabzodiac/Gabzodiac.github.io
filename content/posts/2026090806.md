## 用Node.js接入成本有多低？代码给你看

接入Grok4Fast的核心思想，就是不再直接访问OpenAI的API端点，而是通过云雾的兼容接口做一次“中间处理”。这个中间处理不增加延迟，反而因为国内节点的优化，速度比直连快许多。

下面是一份完整的Node.js示例代码，目标是通过云雾调用GPT-4o，然后处理结果并返回。

javascript
const OpenAI = require('openai');

// 重要改动：把base_url从 "https://api.openai.com/v1" 换成下面这个
const openai = new OpenAI({
  baseURL: 'https://www.yunwuai.cc/v1',
  apiKey: '这里填入你在云雾申请的API Key'
});

async function callGrok4Fast(prompt) {
  try {
    const response = await openai.chat.completions.create({
      model: 'gpt-4o',  // 或者换成你需要的模型
      messages: [
        {
          role: 'user',
          content: prompt
        }
      ],
      max_tokens: 200,
      stream: false  // 改为true即可开启流式输出
    });

    console.log('调用成功，结果:', response.choices[0].message.content);
    return response.choices[0].message.content

  } catch (error) {
    console.error('调用失败:', error);
    throw new Error('接口调用异常，请检查API Key或网络状态');
  }
}

// 调用示例
callGrok4Fast('请用中文介绍Grok4Fast的核心算法原理').then(result => {
  console.log('最终输出:', result);
});
