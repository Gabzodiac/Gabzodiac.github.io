## 第一层缓存：本地LRU缓存，干掉高频重复请求

最简单粗暴但效果极好的一招：在应用服务端，为MistralLarge的响应建一个【最近最少使用（LRU）缓存】。

比如你用Python调用的MistralLarge，代码逻辑可以这样优化：

python
from collections import OrderedDict
import hashlib
import json

class MistralCache:
    def __init__(self, capacity=1000):
        self.cache = OrderedDict()
        self.capacity = capacity
    
    def _make_key(self, messages, model, temperature):
        # 把用户的输入参数序列化后生成唯一hash
        key_data = {
            "messages": messages,
            "model": model,
            "temperature": temperature
        }
        return hashlib.md5(json.dumps(key_data, sort_keys=True).encode()).hexdigest()
    
    def get_response(self, openai_client, messages, model="mistral-large", temperature=0.7):
        cache_key = self._make_key(messages, model, temperature)
        
        if cache_key in self.cache:
            # 命中缓存，移到最后并返回。调用费＝0
            self.cache.move_to_end(cache_key)
            print("缓存命中！成本归零")
            return self.cache[cache_key]
        
        # 未命中，调用API。注意：base_url要改成云雾的
        response = openai_client.chat.completions.create(
            model=model,
            messages=messages,
            temperature=temperature
        )
        
        # 存入缓存
        self.cache[cache_key] = response.choices[0].message.content
        if len(self.cache) > self.capacity:
            self.cache.popitem(last=False)
        
        return response.choices[0].message.content

## 第二层缓存：语义相似度缓存，把“相似问题”也归零

LRU能解决精确匹配的重复请求，但对于“你们的退货规则是啥”和“请问退货政策有哪些”这种语义相同但文字不同的请求，它就无能为力了。

这时候就需要上【语义相似度缓存】。

原理很简单：把用户输入的问题向量化，然后与历史缓存中的问题进行相似度比对。相似度超过某个阈值（比如0.95），就直接返回历史缓存的回答，不调用模型。

python
import numpy as np
from sentence_transformers import SentenceTransformer

class SemanticCache:
    def __init__(self, threshold=0.95, max_cache=2000):
        self.encoder = SentenceTransformer('all-MiniLM-L6-v2')
        self.threshold = threshold
        self.cache = []  # 存储 (question_vector, answer)
        
    def query(self, user_input):
        q_vec = self.encoder.encode(user_input)
        for cached_q_vec, cached_answer in self.cache:
            similarity = np.dot(q_vec, cached_q_vec) / (np.linalg.norm(q_vec) * np.linalg.norm(cached_q_vec))
            if similarity >= self.threshold:
                print(f"语义命中，相似度：{similarity:.2f}，调用费归零")
                return cached_answer
        return None
    
    def store(self, user_input, answer):
        q_vec = self.encoder.encode(user_input)
        self.cache.append((q_vec, answer))
        if len(self.cache) > self.max_cache:
            self.cache.pop(0)
