import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import com.theokanning.openai.service.OpenAiService;
import io.reactivex.Flowable;
import java.time.Duration;
import java.util.Arrays;

public class DeepSeekCoderJavaExample {
    public static void main(String[] args) {
        // 关键步骤：把API Key替换成你在云雾申请的Key
        OpenAiService service = new OpenAiService("Your_YunWuAI_API_KEY", Duration.ofSeconds(30));

        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("DeepSeek-Coder-V2") // 或者 DeepSeek-R1
                .messages(Arrays.asList(
                        new ChatMessage("system", "你是一位顶级的Java开发工程师，精通Spring Boot和微服务架构。"),
                        new ChatMessage("user", "请用Java写一个方法，接收一个字符串列表，返回排序后的新列表，要求忽略大小写。")
                ))
                .maxTokens(4096)
                .temperature(0.5)
                .build();

        // 流式输出
        Flowable<ChatCompletionChunk> flowable = service.streamChatCompletion(request);
        flowable.subscribe(chunk -> System.out.print(chunk.getChoices().get(0).getMessage().getContent()));
    }
}
