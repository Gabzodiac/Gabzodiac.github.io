import requests

def text_to_speech(text_to_speak):
    url = "https://www.yunwuai.cc/v1/audio/speech"  # 核心！这就是新版OpenAI兼容接口地址
    headers = {

    payload = {
        "model": "tts-1-hd-cloudyvoice-r2",  # 推荐云雾自研的高质量中文模型
