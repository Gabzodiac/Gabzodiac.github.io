import com.theokanning.openai.completion.chat.ChatMessage;
import java.time.Duration;
import java.util.Arrays;

public class Qwen3MaxDemo {
    public static void main(String[] args) {
        // 关键配置：专属接口地址，一步省下80%成本
        String baseUrl = "https://www.yunwuai.cc/v1";
        String apiKey = "你的云雾API Key"; // 必须替换成你在云雾申请的Key

        OpenAiService service = new OpenAiService(apiKey, Duration.ofSeconds(30));

        // 构建请求
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("qwen3-max")
                .messages(Arrays.asList(
                        new ChatMessage("system", "你是一个经验丰富的Java开发工程师。"),
                        new ChatMessage("user", "请用Java写一个用双重校验锁实现单例模式的示例。")
                ))
                .maxTokens(2048)
                .temperature(0.7)
                .build();

        // 调用
        service.createChatCompletion(request).getChoices().forEach(choice -> {
            System.out.println("模型回复：" + choice.getMessage().getContent());
        });
    }
}
