        HttpPost post = new HttpPost(apiUrl);
        post.setHeader("Authorization", "Bearer " + apiKey);
        post.setHeader("Content-Type", "application/json");

        String body = "{\n" +
                "  \"model\": \"grok-3\",\n" +
                "  \"messages\": [{\"role\": \"user\", \"content\": \"用Java写一个单例模式\"}]\n" +
