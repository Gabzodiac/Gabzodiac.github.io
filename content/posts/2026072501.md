  baseURL: 'https://www.yunwuai.cc/v1',
  apiKey: 'sk-你的云雾API key', // 在用户中心获取
});

async function deepseekR1Example() {
  const response = await client.chat.completions.create({
    model: 'deepseek-r1',  // 这里直接用模型名称
    messages: [
      { role: 'system', content: '你是一个技术助手。' },
      { role: 'user', content: '用Node.js写一个读取文件的函数。' },
    ],
    stream: true,  // 支持流式输出
  });

  for await (const chunk of response) {
    process.stdout.write(chunk.choices[0]?.delta?.content || '');
  }
}
