以前直连通义千问，用的是阿里云给的域名。现在换成云雾：

import com.theokanning.openai.completion.CompletionRequest;
import com.theokanning.openai.service.OpenAiService;
import okhttp3.OkHttpClient;
import java.time.Duration;

public class MyQwenService {

    public static void main(String[] args) {
        // 最关键的一步：设置base_url
        String token = "你的云雾API Key";
        OpenAiService service = new OpenAiService(token, Duration.ofSeconds(60), 
            new OkHttpClient.Builder().build());

        // 在这里指定模型，要使用通义千问，填“qwen-turbo”或“qwen-plus”等
        CompletionRequest completionRequest = CompletionRequest.builder()
                .model("qwen-turbo") // 云雾兼容通义千问模型名
                .prompt("请用中文写一份Java代码示例，演示如何使用Stream API过滤列表。")
                .maxTokens(2000)
                .build();

        // 调用
        service.createCompletion(completionRequest).getChoices().forEach(choice -> {
            System.out.println(choice.getText());
        });

        // 注意！不要用shutdownExecutor，保持连接池
        // service.shutdownExecutor(); 删除这行
    }
}
