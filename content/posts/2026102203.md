import requests
import json

# [云雾ai中转站](https://www.yunwuai.cc/)的API地址和密钥

def call_ai_model(prompt):
    headers = {
        "Authorization": f"Bearer {api_key}",
        "Content-Type": "application/json"
    }
    data = {
        "model": "gpt-4o",  # 也可以换成 claude-3.5-sonnet 等

    else:

# 模拟用户提问

reply = call_ai_model(user_question)

# 发送到钉钉群
payload = {
    "msgtype": "text",
    "text": {
        "content": f"🤖 AI回复：\n{reply}"
    }
}
requests.post(dingtalk_webhook, json=payload)
